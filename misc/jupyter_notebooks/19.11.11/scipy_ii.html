def f(t, a, b, c):
    return b * np.exp(-a*t) * t**2 + c

def s(y, f, *args):
    residual = (y - f(*args))
    return 0.5 * residual @ residual

_a = 0.75
_b = 2
_c = 0.5
_x = (_a, _b, _c)

_n = 30
t = np.linspace(0, 10, _n)
y = f(t, *_x) + np.random.normal(0, 0.1, _n)

min_res = optimize.minimize(lambda x: s(y, f, t, *x), x0=(1, 1, 1))
print(min_res.x)
ls_res = optimize.least_squares(lambda x: y - f(t, *x), x0=(1, 1, 1),
                                method='lm')
print(ls_res.x)
optimize.curve_fit(f, xdata=t, ydata=y, p0=(1, 1, 1))
